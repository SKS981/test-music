### 2.5 自动发送机制

#### 2.5.1 沉默阈值检查
```javascript
// 检查沉默阈值
async function checkSilenceThresholds() {
  const { data: relics, error } = await supabase
    .from('relics')
    .select('*, contacts(*)')
    .eq('status', 'pending')
    .eq('user_id', currentUserId);

  if (error) {
    console.error('查询遗物失败:', error);
    return;
  }

  const now = new Date();
  relics.forEach(async (relic) => {
    const createdAt = new Date(relic.created_at);
    const thresholdDays = relic.silence_threshold;
    const sendDate = new Date(createdAt.getTime() + thresholdDays * 24 * 60 * 60 * 1000);

    if (now >= sendDate) {
      await sendRelic(relic);
    }
  });
}
```

// 发送遗物
async function sendRelic(relic) {

    // 调用短信发送API（待实现）

### 3.3 交互逻辑

#### 3.3.1 录制控制
```javascript
// 录制状态管理
let mediaRecorder = null;
let recordedChunks = [];
let stream = null;

// 开始录制
async function startRecording() {
  try {
    stream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });
    const videoElement = document.getElementById('preview-video');
    videoElement.srcObject = stream;
    
    mediaRecorder = new MediaRecorder(stream, { mimeType: 'video/webm' });
    recordedChunks = [];
    
    mediaRecorder.ondataavailable = (event) => {
      if (event.data.size > 0) {
        recordedChunks.push(event.data);
      }
    };
    
    mediaRecorder.onstop = async () => {
      const blob = new Blob(recordedChunks, { type: 'video/webm' });
      await uploadRecording(blob);
      // 停止媒体流
      stream.getTracks().forEach(track => track.stop());
    };
    
    mediaRecorder.start();
    console.log('开始录制');
  } catch (error) {
    console.error('录制失败:', error);
  }
}

// 停止录制
function stopRecording() {
  if (mediaRecorder && mediaRecorder.state !== 'inactive') {
    mediaRecorder.stop();
    console.log('停止录制');
  }
}
```

5. **配置策略**：设置 Row Level Security 策略

### 4.2 数据库初始化脚本

```sql
-- 创建用户表
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  email TEXT UNIQUE NOT NULL,
  created_at TIMESTAMP DEFAULT now(),
  last_login TIMESTAMP
);

-- 创建联系人表
CREATE TABLE contacts (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  email TEXT,
  phone TEXT,
  created_at TIMESTAMP DEFAULT now()
);

-- 创建遗物表
CREATE TABLE relics (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  content TEXT,
  silence_threshold INTEGER NOT NULL,
  recipient_id UUID REFERENCES contacts(id) ON DELETE CASCADE,
  anonymous_send BOOLEAN DEFAULT false,
  video_url TEXT,
  audio_url TEXT,
  status TEXT DEFAULT 'pending',
  created_at TIMESTAMP DEFAULT now(),
  sent_at TIMESTAMP
);

-- 创建索引
CREATE INDEX idx_contacts_user_id ON contacts(user_id);
CREATE INDEX idx_relics_user_id ON relics(user_id);
CREATE INDEX idx_relics_status ON relics(status);

-- 启用 RLS
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE contacts ENABLE ROW LEVEL SECURITY;
ALTER TABLE relics ENABLE ROW LEVEL SECURITY;

-- 创建策略
CREATE POLICY "Users can view their own profile" ON profiles
  FOR SELECT USING (auth.uid() = id);

CREATE POLICY "Users can create their own contacts" ON contacts
  FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can view their own contacts" ON contacts
  FOR SELECT USING (auth.uid() = user_id);

CREATE POLICY "Users can create their own relics" ON relics
  FOR INSERT WITH CHECK (auth.uid() = user_id);

CREATE POLICY "Users can view their own relics" ON relics
  FOR SELECT USING (auth.uid() = user_id);
```

## 五、安全与隐私

### 5.1 数据安全

#### 5.1.1 用户认证安全
- 使用 Supabase Auth 进行身份验证
- 支持 JWT token 验证
- 密码使用 bcrypt 加密存储
- 支持邮箱验证和密码重置

#### 5.1.2 数据访问控制
- 使用 Row Level Security 限制数据访问
- 用户只能访问自己的数据
- 敏感操作需要用户认证

#### 5.1.3 文件安全
- 使用 Signed URLs 控制文件访问
- 文件存储在私有存储桶
- 只有授权用户可以访问文件

### 5.2 隐私保护

#### 5.2.1 匿名发送
- 支持匿名发送遗物
- 匿名发送时隐藏发送者信息
- 接收者无法获取发送者的联系方式

#### 5.2.2 数据加密
- 传输使用 HTTPS
- 数据库数据静态加密
- 文件存储加密

#### 5.2.3 用户控制
- 用户可以删除自己的遗物
- 用户可以删除自己的联系人
- 用户可以导出自己的数据
